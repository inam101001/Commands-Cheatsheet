Groovy Script Console Snippets

// List all installed plugins
Jenkins.instance.pluginManager.plugins.each {
  println "${it.shortName}: ${it.version}"
}
// Cancel all items in the build queue
Jenkins.instance.queue.clear()
// Safely clean up workspace of all projects
for (item in Jenkins.instance.getAllItems(AbstractProject)) {
  item.doWipeOutWorkspace()
}
// Enable/Disable all jobs
Jenkins.instance.getAllItems(Job).each { job ->
  job.disabled = true
  job.save()
}

Declarative Pipeline Basics

pipeline {
    agent any
    environment {
        APP_ENV = 'production'
    }
    options {
        timeout(time: 1, unit: 'HOURS')
        ansiColor('xterm')
    }
    stages {
        stage('Build') {
            steps {
                echo 'Building...'
            }
        }
    }
    post {
        always {
            cleanWs()
        }
    }
}

Scripted Pipeline & Steps

node('linux-agent') {
    try {
        stage('Checkout') {
            checkout scm
        }
        stage('Build') {
            sh './gradlew build'
        }
    } catch(err) {
        currentBuild.result = 'FAILURE'
        throw err
    } finally {
        echo "Cleaned up."
    }
}

Credentials Management

// Use Username/Password
withCredentials([usernamePassword(
    credentialsId: 'my-credentials-id',
    usernameVariable: 'USER',
    passwordVariable: 'PASS')]) {
    sh 'curl -u $USER:$PASS http://api.com'
}
// Use Secret Text
withCredentials([string(
    credentialsId: 'my-token-id',
    variable: 'API_TOKEN')]) {
    sh 'curl -H "Auth: $API_TOKEN" URL'
}

Docker in Pipelines

// Run inside a Docker container
agent {
    docker { image 'node:20-alpine' }
}

// Inline Docker usage (Scripted)
docker.image('maven:3-openjdk-17').inside {
    sh 'mvn clean install'
}
// Build & Push Docker Image
docker.withRegistry('https://ghcr.io', 'registry-creds') {
    def img = docker.build("my-app:${env.BUILD_NUMBER}")
    img.push()
}

Build Triggers & Parameters

// Triggers block
triggers {
    cron('H 4 * * 1-5') // Mon-Fri 4 AM
    pollSCM('H/15 * * * *') // Poll SCM
    upstream(upstreamProjects: 'BuildAll', threshold: hudson.model.Result.SUCCESS)
}
// Parameters block
parameters {
    string(name: 'DEPLOY_ENV', defaultValue: 'dev', description: 'Target env')
    booleanParam(name: 'RUN_TESTS', defaultValue: true)
    choice(name: 'BROWSER', choices: ['Chrome', 'Firefox', 'Safari'])
}
